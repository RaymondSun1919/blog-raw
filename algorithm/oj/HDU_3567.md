#### HDU 3567: Eight II
> refer [http://acm.hdu.edu.cn/showproblem.php?pid=3567](http://acm.hdu.edu.cn/showproblem.php?pid=3567)

关键字：IDA*，康拓展开，曼哈顿距离


```cpp
#include <bits/stdc++.h>

using namespace std;

char dc[] = {'d', 'l', 'r', 'u'};
int dx[] = {1, 0, 0, -1};
int dy[] = {0, -1, 1, 0}; 

int radix[] = {1, 1, 2, 6, 24, 120, 720, 5040, 40320, 362880};

/**
 * @brief 计算康拓展开
 * 
 * */
int Contar(int *t){
    int ret = 0;
    for(int i = 0; i < 9; ++i){
        int tmp = 0;
        for(int j = i + 1; j < 9; ++j){
            if(t[i] > t[j]){
                ++tmp;
            }
        }
        ret += radix[8 - i] * tmp;
    }
    return ret;
}

bool check(int tx, int ty){
    return tx >= 0 and tx < 3 and ty >= 0 and ty < 3;
}
int src[9];
int dst[9];
int st, ed;
int vis[370000];
char path[370000];
int dstx[9];
int dsty[9];

/**
 * @brief 计算曼哈顿距离
 * 
 * */
int dis(int* t){
    int ans = 0;
    for(int i = 0; i < 9; ++i){
        if(t[i]){
            ans += (abs(i / 3 - dstx[t[i]]) + abs(i % 3 - dsty[t[i]]));
        }
    }
    return ans;
}
int app, maxLen;
/**
 * @brief 迭代加深搜索
 * 
 * */
int IDAstar(int dep, int cur, int curx, int cury){
    if(cur == ed){
        return dep;
    }
    for(int i = 0; i < 4; ++i){
        int nx = curx + dx[i];
        int ny = cury + dy[i];
        if(check(nx, ny)){
            swap(src[nx * 3 + ny], src[curx * 3 + cury]);
            int tmpDis = dep + 1 + dis(src);
            if(tmpDis <= maxLen){
                int tmpCon = Contar(src);
                if(!vis[tmpCon]){
                    vis[tmpCon] = 1;
                    path[dep] = dc[i];
                    int ret = IDAstar(dep + 1, tmpCon, nx, ny);
                    if(ret >= 0){
                        return ret;
                    }
                    vis[tmpCon] = 0;
                }
            }else{
                app = min(app, tmpDis);
            }
            swap(src[nx * 3 + ny], src[curx * 3 + cury]);
        }
    }
    return -1;
}
int main(){
    int T;
    char str[12];
    scanf("%d", &T);
    for(int kase = 1; kase <= T; ++kase){
        int sx, sy;
        scanf("%s", str);
        for(int i = 0; i < 9; ++i){
            if(isdigit(str[i])){
                src[i] = str[i] - '0';
            }else{
                src[i] = 0;
                sx = i / 3;
                sy = i % 3;
            }
        }
        st = Contar(src);
        scanf("%s", str);
        for(int i = 0; i < 9; ++i){
            if(isdigit(str[i])){
                dst[i] = str[i] - '0';
            }else{
                dst[i] = 0;
            }
            dstx[dst[i]] = i / 3;
            dsty[dst[i]] = i % 3;
        }
        ed = Contar(dst);
        memset(vis, 0, sizeof vis);
        vis[st] = 1;
        for(maxLen = 1; ;maxLen = app){
            app = 0x3f3f3f3f;
            int ret = IDAstar(0, st, sx, sy);
            if(ret >= 0){
                printf("Case %d: %d\n", kase, ret);
                path[ret] = '\0';
                puts(path);
                break;
            }
        }
    }
    return 0;
}
```
