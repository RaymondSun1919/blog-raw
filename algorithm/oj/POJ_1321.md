
#### POJ 1321: 棋盘问题
> [http://poj.org/problem?id=1321](http://poj.org/problem?id=1321)

##### 示例代码
```cpp
/**
 * @ref http://poj.org/problem?id=1321
 * @brief 棋盘问题
 * @date 2021/05/30
 * @author dianhsu
 * */
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
char str[10][10];
int n, m;
int ans;
int vis[10];
/**
 * @param dep 当前遍历的深度
 * @param cnt 当前放下的棋子的数目
 * @note 一层一层地遍历，每一层是'#'的地方可以放下棋子，也有可能当前层一个棋子都不放。
 **/
void dfs(int dep, int cnt){
    if(cnt >= m){
        ++ans;
        return;
    }
    if(dep >= n){
        return;        
    }
    for(int i = 0; i < n; ++i){
        if(str[dep][i] == '#' and vis[i] == 0){
            vis[i] = 1;
            dfs(dep + 1, cnt + 1);
            vis[i] = 0;
        }
    }
    // 当前层不放棋子。
    dfs(dep + 1, cnt);
}

int solve(){
    ans = 0;
    memset(vis, 0, sizeof vis);
    dfs(0, 0);
    return ans;
}
int main() {
    while (scanf("%d%d", &n, &m) != EOF) {
        if(n == -1){
            break;
        }
        for(int i = 0; i < n; ++i){
            scanf("%s", str[i]);
        }
        printf("%d\n", solve());
    }
    return 0;
}
```
