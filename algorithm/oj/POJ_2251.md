
#### POJ 2251: Dungeon Master
> [http://poj.org/problem?id=2251](http://poj.org/problem?id=2251)

##### 简要思路
这是一个三维的地形。在每个位置，下一个步骤可以走的方向有六个：前后左右上下。于是方向数组如下所示：
```cpp
int dr[6][3] = {{0, 0, 1}, {0, 0, -1}, {0, 1, 0}, {0, -1, 0}, {1, 0, 0}, {-1, 0, 0}};
```

从某个结点遍历它后面的六个方向的结点的时候，我们需要考虑：后面的结点是否在棋盘上；后面的结点是否已经被访问过了；后面的结点是否是可以到达的结点而不是墙壁。

另外，由于POJ并不支持C++11标准，所以构造一个新的`Point`的时候，并不能用列表构造，只能写了一个构造函数，方便后面使用。

##### 示例代码

```cpp
/**
 * @brief Dungeon Master
 * @date 2021/05/30
 * @author dianhsu
 * @ref http://poj.org/problem?id=2251
 * */

#include <cstdio>
#include <cstring>
#include <queue>
#include <algorithm>
#include <cmath>
using namespace std;

int L, R, C;
char board[33][33][33];
int sx, sy, sz, ex, ey, ez;
int vis[33][33][33];
struct Point {
    int x, y, z;
    int cost;
    Point(int x, int y, int z, int cost) : x(x), y(y), z(z), cost(cost) {

    }
};

int dr[6][3] = {{0, 0, 1}, {0, 0, -1}, {0, 1, 0}, {0, -1, 0}, {1, 0, 0}, {-1, 0, 0}};
// queue
queue<Point> Q;
int solve() {
    while (!Q.empty()) Q.pop();
    Q.push(Point(sx, sy, sz, 0));
    memset(vis, 0, sizeof vis);
    vis[sx][sy][sz] = 1;
    while (!Q.empty()) {
        Point p = Q.front();
        Q.pop();
        for (int i = 0; i < 6; ++i) {
            int nx = p.x + dr[i][0];
            int ny = p.y + dr[i][1];
            int nz = p.z + dr[i][2];
            if (nx >= 0 and nx < L and
                ny >= 0 and ny < R and
                nz >= 0 and nz < C and
                board[nx][ny][nz] != '#' and
                vis[nx][ny][nz] == 0) {
                if (nx == ex and ny == ey and nz == ez) {
                    return p.cost + 1;
                }
                vis[nx][ny][nz] = 1;
                Q.push(Point(nx, ny, nz, p.cost + 1));
            }
        }
    }
    return -1;
}

int main() {
    while (scanf("%d%d%d", &L, &R, &C) != EOF) {
        if (L == 0) break;
        for (int i = 0; i < L; ++i) {
            for (int j = 0; j < R; ++j) {
                scanf("%s", board[i][j]);
                for (int k = 0; k < C; ++k) {
                    if (board[i][j][k] == 'S') {
                        sx = i;
                        sy = j;
                        sz = k;
                    } else if (board[i][j][k] == 'E') {
                        ex = i;
                        ey = j;
                        ez = k;
                    }
                }
            }
        }
        int ret = solve();
        if (ret == -1) {
            printf("Trapped!\n");
        } else {
            printf("Escaped in %d minute(s).\n", ret);
        }
    }
    return 0;
}
```
