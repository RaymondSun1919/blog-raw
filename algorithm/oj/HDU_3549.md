#### HDU 3549: Flow Problem
> refer [http://acm.hdu.edu.cn/showproblem.php?pid=3549](http://acm.hdu.edu.cn/showproblem.php?pid=3549)

**关键词**：网络流，最大流，链式前向星，Edmond-Karp

##### 简要思路

网络流的第一道题，适合学习网络流的使用。

##### 参考代码

```cpp
/**
 * 网络流 Edmond-Karp
 *
 * 链式前向星
 *
 * */
#include <bits/stdc++.h>

using namespace std;
const int inf = 0x3f3f3f3f;
const int maxn = 20;
const int maxm = 2020;

int ans[maxn];
int head[maxn];
int u[maxm], v[maxm], pre[maxm], cap[maxm], flow[maxm];
int edge_cnt;
int p[maxn];
void init() {
    memset(head, -1, sizeof head);
    edge_cnt = 0;
}
void add_edge(int a, int b, int c) {
    // 添加正向边
    u[edge_cnt] = a;
    v[edge_cnt] = b;
    cap[edge_cnt] = c;
    flow[edge_cnt] = 0;
    pre[edge_cnt] = head[a];
    head[a] = edge_cnt++;
    // 添加反向边
    u[edge_cnt] = b;
    v[edge_cnt] = a;
    cap[edge_cnt] = 0;
    flow[edge_cnt] = 0;
    pre[edge_cnt] = head[b];
    head[b] = edge_cnt++;
}
int EdmondKarp(int s, int t) {
    int ret = 0;
    while (1) {
        memset(ans, 0, sizeof ans);
        queue<int> Q;
        ans[s] = inf;
        Q.push(s);
        while (!Q.empty()) {
            int it = Q.front();
            Q.pop();
            for (int i = head[it]; i != -1; i = pre[i]) {
                if (!ans[v[i]] and cap[i] > flow[i]) {
                    p[v[i]] = i;
                    ans[v[i]] = min(ans[it], cap[i] - flow[i]);
                    Q.push(v[i]);
                }
            }
            if (ans[t]) break;
        }
        if (!ans[t]) break;
        for (int i = t; i != s; i = u[p[i]]) {
            flow[p[i]] += ans[t];
            flow[p[i] ^ 1] -= ans[t];
        }
        ret += ans[t];
    }
    return ret;
}
int n, m;
int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);
    int T;
    cin >> T;
    int kase = 0;
    while (T--) {
        init();
        cin >> n >> m;
        int a, b, c;
        for (int i = 0; i < m; ++i) {
            cin >> a >> b >> c;
            add_edge(a, b, c);
        }
        cout << "Case " << ++kase << ": " << EdmondKarp(1, n) << endl;
    }
    return 0;
}
```
